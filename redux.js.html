import { choose } from "./utils"




/**
 * Create clean and readable reducers for redux.
 *
 * @function createReducer
 * @param {Object} [initState={}]
 * @returns {Function}
 */
export const createReducer = (initState = {}) =>
    (actions, defaultAction = (s, _a) => s) =>
        (state = initState, action) =>
            choose(
                action.type,
                actions,
                defaultAction,
                [state, action,]
            )
